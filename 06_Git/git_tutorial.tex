	No Git, quase tudo pode ser feito \bgreen{localmente} \pause
\begin{columns}

\end{center}
\end{column}
\begin{column}[c]{6.5cm}
\pause
\begin{block}{Um fluxo de trabalho}
\begin{enumerate}
\item \textbf{add} inclui no índice arquivos {\bf \tred{modificados}} da {\bf área de trabalho} 
  \begin{itemize}
  \item os arquivos estão {\bf \tred{staged}}
  % Se um arquivo em algum momento esteve staged e não foi explicitamente removido, ele 
  % também é tracked pelo repositório. Isso significa que o Git não irá ignorá-lo e que
  % modificações a ele serão notadas (embora ele só se torne staged novamente caso 
  % o comando git add seja utilizado de novo). 
  \item arquivos {\bf \tred{não-modificados}} não são incluídos
  % Se um arquivo está staged, isso significa que ele está registrado no índice e pronto
  % para ser confirmado no repositório. 
  \end{itemize}
\pause
\item \textbf{commit} inclui no {\bf repositório} itens que estão no {\bf índice}
  \begin{itemize}
  \item os arquivos estão {\bf \tred{confirmados}}
  \end{itemize}
\pause
\item \textbf{checkout} faz duas coisas:
  \begin{itemize}
  \item Seleciona um outro \textit{branch} 
  \item Atualiza a área de trabalho com os arquivos do \textit{branch} selecionado
  \end{itemize}
\end{enumerate}
\end{block}
\end{column}
\end{columns}
\end{frame}

\end{column}
\begin{column}[c]{6.5cm}
\pause
\begin{block}{Um outro fluxo de trabalho}
\begin{enumerate}
\item \textbf{push} envia modificações para o {\bf repositório remoto}
  \begin{itemize}
  \item para os \textit{branches} especificados 
  \end{itemize}
\pause
\item \textbf{fetch} atualiza o repositório local a partir de um remoto
  \begin{itemize}
  \item sem atualizar a area de trabalho
  \end{itemize}
\pause
\item \textbf{pull} é similar a {\tt fetch}, mas atualiza a área de trabalho
\end{enumerate}
\end{block}
\end{column}
\end{columns}
\end{frame}
