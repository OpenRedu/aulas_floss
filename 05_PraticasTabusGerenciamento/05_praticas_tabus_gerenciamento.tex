\begin{frame}
	\frametitle{Sem coerção (mas com alguma hierarquia)}
    \begin{block}{Hierarquia baseada em mérito}
    \begin{enumerate}
      \item O {\bf \bgreen{criador}} do projeto é um caso especial\vspace{0.1cm}
      \item \textit{Committers} e não-\textit{committers}\vspace{0.1cm}
      \begin{itemize}
         \item Às vezes há vários níveis de \textit{committers}
      \end{itemize}
    \end{enumerate}
    \end{block}\pause

    \begin{block}{``Norma sem sanção é como fogo que não queima''?} 
    % Esse dito é atribuído a Rudolf von Jhering (http://en.wikipedia.org/wiki/Rudolf_von_Jhering)
    \begin{enumerate}
      \item Projetos \textit{open source} têm regras
         \begin{itemize}
         \item Como fazer que sejam seguidas?\vspace{0.1cm}
         \end{itemize}
         \pause
      \item Ditador \tred{{\em benevolente}}?
      % Em um projeto FLOSS, o ditador funciona muito mais como um mediador de
      % conflitos do que como um ditador de fato. Como discutido a seguir, o 
      % motivo para isso é a possibilidade sempre presente de se criar um fork. 
      % Em outras palavras, o ditador não tem nenhum poder de coerção e não
      % pode ser responsabilizado por manter a ordem e garantir que as regras
      % sejam cumpridas. 
      \begin{itemize}
      \item Tem que ser, senão, enFORKa ele\vspace{0.1cm}
      \end{itemize}
      \pause
      \item Reputação é {\bf tudo}
      % De acordo com Eric Raymond em "Homesteading the Noosphere":
      % ...Software is freely shared. This abundance creates a situation in which the 
      % only available measure of competitive success is reputation among one's peers.
         \begin{itemize}
         \item E as regras costumam ser pragmáticas
         % Como consequência desse pragmatismo, não seguir as regras tem efeitos 
         % negativos práticos e observáveis nos outros participantes do projeto. 
         % Logo, não seguir as regras pode colocar em cheque a competência de um
         % participante, o que pode arranhar sua reputação. Como reputação é tudo,
         % desvios das regras são evitados a todo custo. 
         \end{itemize}
    \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}
  \frametitle{A importância da \textit{fork}abilidade}
 {\large
  \begin{itemize}
  	\item \textit{Forks} são a \tred{cola fundamental} de um projeto de FLOSS
      \begin{itemize}
	  \item Mantém a balança de poder\vspace{0.1cm}
	  \item {\bf \tred{Toda}} licença de FLOSS permite a criação de \textit{forks}
	  \end{itemize}\vspace{0.4cm}
\pause
	  \item Democratizam ditaduras
        \begin{itemize}
        \item Poder é {\bf dado} ao ``ditador''
	    \end{itemize}\vspace{0.4cm}
	  \item Reduzem as vantagens de fechar o código
        \begin{itemize}
        \item Quando uma corporação compra uma outra
	    \end{itemize}
	  
	  % Segundo Karl Fogel: 
	  % Forks, or rather the potential for forks, are the reason there are no true 
	  % dictators in free software projects. This may seem like a surprising claim, 
	  % considering how common it is to hear someone called the "dictator" or "tyrant" 
	  % in a given open source project. But this kind of tyranny is special, quite 
	  % different from the conventional understanding of the word. Imagine a king 
	  % whose subjects could copy his entire kingdom at any time and move to the 
	  % copy to rule as they see fit. Would not such a king govern very differently 
	  % from one whose subjects were bound to stay under his rule no matter what he 
	  % did?
	  %

  \end{itemize}

    \item LibreOffice (\byellow{OpenOffice})\vspace{0.2cm}\pause

    % que o rebatizou de Apache OpenOffice.
    \item EGCS -- Enhanced/Experimental GNU Compiler System (\byellow{GCC})\vspace{0.2cm}\pause
    % EGCS = The Enhanced/Experimental GNU Compiler System
    % Motivo: devido à grande popularidade do GCC, a FSF tornou-se muito preocupada com a 
    % estabilidade do mesmo, o que significava que a inclusão de novas funcionalidades tornou-se
    % menos prioritária e mais cuidadosa, para evitar prejudicar essa estabilidade. Alguns
    % desenvolvedores, ansiosos por introduzir melhorias no tronco do código do compilador, 
    % várias delas já implementadas ao longo dos anos, resolveram criar o EGCS em 1997, que 
    % inicialmente servia para experimentar com novas funcionalidades ainda não consideradas 
    % maduras o suficiente para entrar no GCC. Dizer que o EGCS é um fork não é inteiramente 
    % preciso. Na verdade, ele é o resultado da integração de diversos forks do GCC. Em 1999 
    % tornou-se o GCC oficial. O GCC até então era desenvolvido em um estilo "Catedral". A 
    % partir daí, passou a empregar a abordagem "Bazar", já usada no desenvolvimento do EGCS.
    
    \item Apache HTTPd (\byellow{NCSA HTTPd})\vspace{0.2cm}\pause
    % Motivo: O desenvolvimento do NCSA HTTPd ia muito lento na segunda metade da década de 90. Um
    % grupo de desenvolvedores pegou o código do servidor e criou um fork, o Apache HTTPd. 
    % Hoje em dia, o Apache não inclui mais código originalmente do NCSA HTTPd.  
        
    \item Drizzle (\byellow{MySQL})\vspace{0.2cm}\pause
    % Motivo: A Oracle, com a compra da Sun, tornou-se dona do banco de dados comercial mais usado
    % no mundo e do banco de dados livre mais usado no mundo. Alguns desenvolvedores consideraram
    % essa situação menos que ideal e criaram um fork. 
    
    \item GNU IceCat (\byellow{Mozilla Firefox}) \vspace{0.2cm}\pause
    % Rebranding do Firefox, mas mantido atualizado junto à base de código principal. A Mozilla
    % exigiu que o browser usado no Debian mudasse de nome. Foi criado então o IceCat. Vale
    % ressaltar que o código do IceCat aproveita todas as funcionalidades e melhorias aplicadas
    % ao Firefox, mas inclui código adicional.

\begin{frame}
	\frametitle{Antes, uma palavrinha sobre terminologia...}
	\begin{block}{Projeto}
	\begin{quote}
	``A project is a temporary endeavor with a defined beginning and end (\textbf{usually time-constrained, and often constrained by funding or deliverables}), undertaken to meet unique goals and objectives [...].''
	\end{quote}
	% Definição retirada da Wikipédia (http://en.wikipedia.org/wiki/Project_management)
	% A página da Wikipédia sobre ``Project'' refere-se a projetos de Engenharia, Ciência e Tecnologia (escolas e universidades),
	% projetos conforme definidos pela área de Gerenciamento de Projetos. Em todos os casos, um projeto tem duração finita. 
	% A atividade ``do dia-a-dia'' de uma empresa, que não tem um fim, é chamada de operação.
    \begin{itemize}\pause
    \item \textbf{Estritamente}, um ``projeto'' FLOSS não é um projeto
    % Talvez possa ser visto como um conjunto de projetos.
    \end{itemize}
    \end{block}
\pause
	\begin{block}{Gerenciamento de projetos}
	Responsabilidades de um gerente de projeto: 
	\begin{itemize}
	\item Definição de objetivos (viáveis e claros) para o projeto
    \item Gerenciamento de recursos 
    \item \textbf{\tred{Construção}} dos requisitos do projeto
    % Se usarmos o termo ``construção'', acho que essa definição não é ideal para descrever as atividades do gerente de
    % um projeto FLOSS. Também não sei se ``elicitação'' e ``priorização'' são bons. O termo ``priorização'' me parece
    % o mais preciso.  
    %Da Wikipedia: ``A project manager is the person responsible for accomplishing the stated project objectives.'' Key project management responsibilities include creating clear and attainable project objectives, building the project requirements, and managing the constraints of the project management triangle, which are cost, time, scope, and quality''.
    % http://en.wikipedia.org/wiki/Project_manager#Overview
	\end{itemize}
    \end{block}

\begin{frame}
	\frametitle{Ditador benevolente (BD)}
	\begin{block}{O que é?}
    \begin{itemize}
    \item Autoridade final na tomada de decisões\vspace{0.1cm}
    \item Um \tred{juiz} \textbf{aprovado pela comunidade}\vspace{0.1cm}
    \item Muitas vezes é o fundador do projeto
    % É possível perceber que, da mesma forma que um ``projeto'' de FLOSS não é um projeto estritamente
    % falando, o BD também não é um gerente de projeto, embora haja atribuições comuns. 
    \end{itemize}
    \end{block}
\pause
	\begin{block}{Quem pode ser?}
    \begin{itemize}
    \item Desenvolvedor experiente\vspace{0.1cm}
      \begin{itemize}
      \item Não necessariamente o melhor \textit{hacker}
      \end{itemize}
    \item {\em Principal habilidade}: reconhecer \bred{boas soluções de projeto}\vspace{0.1cm}
    \item Via de regra, é óbvio quem {\bf deve} ser o BD   
    \end{itemize}
    \end{block}

\begin{frame}
	\frametitle{Democracia baseada em consenso (com ou sem comitê)}
	\begin{block}{Funcionamento básico}
    \begin{itemize}
    \item Decisão a ser tomada é apresentada publicamente\vspace{0.1cm}
    \item Se não houver objeções, consenso é atingido\vspace{0.1cm}
    % No Apache HTTPD, há um comitê chamado Apache HTTP Project Management Committee (PMC, for short). Esse comitê é
    % quem faz as votações. Para que qualquer 
    % patch seja committed, é necessário que haja três votos a favor de membros do PMC e nenhum contra. Mais informações
    % no endereço http://httpd.apache.org/ABOUT_APACHE.html
    \item Decisão pode ser implementada\vspace{0.1cm}
    % ``implementada'' em um sentido mais amplo. Não precisa ser uma decisão diretamente
    % ligada ao código. 
    \end{itemize}
    \end{block}
\pause
	\begin{block}{Complementarmente}
    \begin{itemize}
    \item Algumas decisões podem ser tomadas sem discussão\vspace{0.1cm}
      \begin{itemize}
      \item Por exemplo, a decisão de corrigir um \textit{bug}
      \end{itemize}
    \item Consequências podem ser {\bf desfeitas}\vspace{0.1cm}
    % A possibilidade de desfazer as consequências de uma má decisão, possível graças ao
    % sistema de controle de versões, agiliza o processo de tomada de decisão e evita
    % que muito tempo seja gasto propondo publicamente cursos de ação óbvios e antecipando 
    % todas as formas como uma decisão pode não ser boa.
    \item Se consenso não for atingido, é necessário \byellow{votar}  
    \end{itemize}
    \end{block}
\end{frame}

\begin{frame}
	\frametitle{Votação}
	\begin{block}{Quem pode votar?}
    \begin{itemize}\pause
    \item Nem todo mundo\vspace{0.1cm} \pause 
    \item Resposta fácil: \textit{committers}\vspace{0.1cm}\pause
    \item Às vezes apenas um subconjunto.
    % É o casos projetos da Apache Software Foundation e seus PMCs.
    \end{itemize}
    \end{block}
\pause
	\begin{block}{Complicações}
    \begin{itemize}
    \item \textit{Committers} parciais\vspace{0.1cm}
    % ``In projects that offer both full and partial commit access, the question of whether partial committers can vote largely depends on the process by which partial commit access is granted. If the project hands it out liberally, for example as a way of maintaining many third-party contributed tools in the repository, then it should be made clear that partial commit access is really just about committing, not voting.''. Capítulo 4 de Producing Open Source Software, de Karl Fogel. Licença: Creative Commons Attribution-ShareAlike 3.0 Unported.
      \begin{itemize}
      \item \textit{committing} == \textit{voting}?
      \end{itemize}
    \item Vetos\vspace{0.1cm}
    % Um veto é uma objeção forte contra uma decisão. Funciona como uma maneira de protelar uma votação para que mais discussão sobre
    % um determinado tópico aconteça. É feito por uma pessoa com poder de votar. O impacto de um veto varia de projeto para projeto.
    % Desenvolvedores devem usar vetos de maneira parcimoniosa. O veto não é um mecanismo para que desenvolvedores atrapalhem
    % votações para tentar impor sua visão de qual curso de ação de ser tomado. 
    \end{itemize}
    \end{block}

    Projetos maduros definem \textbf{diretrizes} para várias questões, incluindo votações

% A checklist foi extraída a partir do Capítulo 2 do livro "Producing Open Source Software", de Karl Fogel.
\begin{frame}
	\frametitle{Para começar um novo projeto}
    \begin{itemize}
    \item Escolha um nome \vspace{0.1cm}
    \item Defina objetivos de longo prazo\vspace{0.1cm}
    \item Deixe claro que o projeto é livre\vspace{0.1cm}
    % Sobre o ponto acima, um estudo interessante é este: 
    % http://www.theregister.co.uk/2013/04/18/github_licensing_study/
    % Em resumo, como a maior parte dos projetos não deixa claro qual é a licença, esses projetos não são, 
    % legalmente falando, de código aberto. 
    \item Elenque os requisitos e funcionalidades\vspace{0.1cm}
    \item Torne o programa baixável em formato fonte \vspace{0.1cm}
    \item Estabeleça acesso a controle de versões e rastreador de bugs\vspace{0.1cm}
      % Vários sítios podem ser usados para este fim. Atualmente, o Github é bastante
      % popular por combinar repositório com controle de versões baseado no Git, bug
      % tracking e rede social em um só sítio. 
    \item Estabeleça canais de comunicação (lista, canal de IRC)\vspace{0.1cm}
    % De forma geral, meios para que pessoas possam se comunicar e cooperar.
    \item Escolha a licença\vspace{0.1cm}
    \item \textbf{Anuncie}!
    % Em http://www.freecode.com
    \end{itemize}
\end{frame}
